    b.amount 'Commission Amount',
    b.commission_status 'COMMISSIONSTATUS',

		select case 
			when b.bill_type = 'DIRECT_COMM' then 'SC'
			when b.bill_type = 'OR_COMM' then 'OSC'

			when JSON_EXTRACT(fee_details->>'$.jsonGLAttributes','$.lifeOrGI')='GI' then (
